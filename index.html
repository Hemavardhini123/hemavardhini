# 1. Initialize Git inside your project folder git init # 2. Add remote repository (replace with your GitHub repo link) git remote add origin https://github.com/your-username/your-username.github.io.git # 3. Add all files (index.html, CSS, etc.) git add . # 4. Commit the files with a message git commit -m "Initial commit" # 5. Push files to GitHub main branch git branch -M main git push -u origin main
